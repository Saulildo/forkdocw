local function printTable(t, indent)
    indent = indent or ""
    if type(t) ~= "table" then
        print(indent .. tostring(t))
    else
        for k, v in pairs(t) do
            if type(v) == "table" then
                print(indent .. tostring(k) .. ":")
                printTable(v, indent .. "  ")
            else
                print(indent .. tostring(k) .. ": " .. tostring(v))
            end
        end
    end
end
